% \emph does not toggle shape when already in italic/slanted context
\makeatletter
\DeclareRobustCommand{\emph}[1]{%
  \ifdim\fontdimen\@ne\font>\z@
    {#1}%
  \else
    {\itshape #1}%
  \fi}
\makeatother

\definecolor{linkcolor}{HTML}{0a58ca} % #0a58ca
\definecolor{citecolor}{HTML}{59359a} % #59359a
\definecolor{backrefcolor}{HTML}{ab296a} % #ab296a

\AtBeginDocument{\hypersetup{bookmarksdepth=2,linkcolor=linkcolor,citecolor=citecolor,urlcolor=linkcolor}}

\DeclareListFormat{pageref}{%
  \ifthenelse{\value{listcount}>1}{,\addspace}{}%
  \ifhyperref{\hyperlink{page.#1}{\textcolor{backrefcolor}{#1}}}{\textcolor{backrefcolor}{#1}}}

\renewbibmacro*{pageref}{%
  \iflistundef{pageref}
  {}
  {\space$\to$\addspace\printlist[pageref]{pageref}}}

\AtBeginDocument{%
  \newtheoremstyle{plain}
  {\topsep}{\topsep}{\normalfont}{}{\bfseries}{.}{5pt plus 1pt minus 1pt}{}%
}

% Force all TOC entries to use the sans-serif font via tocbasic hooks,
% which fire around the actual entry rendering (after \normalfont resets)
\makeatletter
\BeforeStartingTOC[toc]{%
  \let\toc@@save@@normalfont\normalfont
  \renewcommand{\normalfont}{\toc@@save@@normalfont\sffamily\fontsize{12}{15}\selectfont}%
  \sffamily\fontsize{12}{15}\selectfont
}
\AfterStartingTOC[toc]{\let\normalfont\toc@@save@@normalfont}
\makeatother

\makeatletter
\newif\if@afterpart

% Remove the \newpage from the part ending, set flag instead
\renewcommand*{\partheadendvskip}{%
  \vskip\@tempskipa
  \global\@afterparttrue
}
\makeatother

\makeatletter
\newcommand{\@new@epigraph}[2]{%
  \if@afterpart
    \enlargethispage{-10\baselineskip}%
    \vspace{10\baselineskip}%
  \fi
  \@orig@epigraph{#1}{#2}%
  \if@afterpart
    \vfill%
    \global\@afterpartfalse
  \fi
}
\AtBeginDocument{%
  \let\@orig@epigraph\epigraph
  \let\epigraph\@new@epigraph
}
\makeatother
